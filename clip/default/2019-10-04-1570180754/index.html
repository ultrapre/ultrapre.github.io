set fso=createobject("scripting.filesystemobject")
set f1=fso.opentextfile("script.js", 1)
strfile1=f1.readall
f1.close
strtemp=replace(strfile1,"""","")
strtemp=replace(strtemp,"imagelist:{","")
strtemp=replace(strtemp,"{","")
strtemp=replace(strtemp,"}","")
set fs=createobject("scripting.filesystemobject")
varsplit1=split(strtemp,",")
for each strline in varsplit1
if instr(strline, ":")>0 then
varsplit2=split(strline,":")
if instr(varsplit2(1), "/image/")>0 then
if instr(varsplit2(0), ".png")>0 or instr(varsplit2(0), ".jpg")>0 then
if fso.fileexists(varsplit2(1)) and not fso.fileexists(varsplit2(0)) then
fs.movefile varsplit2(1),"./image/"&varsplit2(0)
end if
elseif instr(varsplit2(1), ".po2")>0 then
strtemp="./image/"&varsplit2(0)&".png"
if fso.fileexists(varsplit2(1)) and not fso.fileexists(strtemp) then
fs.movefile varsplit2(1),strtemp
end if
elseif instr(varsplit2(1), ".jo2")>0 then
strtemp="./image/"&varsplit2(0)&".jpg"
if fso.fileexists(varsplit2(1)) and not fso.fileexists(strtemp) then
fs.movefile varsplit2(1),strtemp
end if
elseif instr(varsplit2(1), ".go2")>0 then
strtemp="./image/"&varsplit2(0)&".gif"
if fso.fileexists(varsplit2(1)) and not fso.fileexists(strtemp) then
fs.movefile varsplit2(1),strtemp
end if
end if
end if
end if
next
wscript.echo "done!"
